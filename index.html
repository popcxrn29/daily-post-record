async function startRandom() { const btn = document.querySelector('.btn-random'); const dateVal = document.getElementById("date").value; if (!dateVal) { alert("กรุณาเลือกวันที่ก่อนครับ"); return; } try { btn.disabled = true; btn.innerText = "กำลังจัดเตรียมทรัพย์..."; // 1. กำหนดทรัพย์หลักที่ต้องมีทุกวัน const fixAssets = [ "บ้านเอื้อ 214/278", // หรือสุ่มจากกลุ่มบ้านเอื้อก็ได้ "แฮปปี้คอนโด" ]; // 2. ดึงคิวทรัพย์ทั้งหมด เพื่อหาทรัพย์ตัวที่ 3 let qSnap = await db.collection("settings").doc("queue").get(); let currentQueue = (qSnap.exists && qSnap.data().list.length > 0) ? qSnap.data().list : [...allAssets]; // กรองเอาทรัพย์หลักออกก่อนเพื่อไม่ให้สุ่มตัวที่ 3 มาซ้ำกับตัวที่ล็อคไว้ let availablePool = currentQueue.filter(item => !fixAssets.includes(item)); // ถ้าคิวหมด ให้รีเซ็ตใหม่ if (availablePool.length === 0) { availablePool = allAssets.filter(item => !fixAssets.includes(item)); } // 3. สุ่มทรัพย์ตัวที่ 3 มา 1 อย่าง let randomIndex = Math.floor(Math.random() * availablePool.length); let thirdAsset = availablePool.splice(randomIndex, 1)[0]; // รวมเป็น 3 ทรัพย์ (2 ตัวล็อค + 1 ตัวสุ่ม) let chosen = [fixAssets[0], fixAssets[1], thirdAsset]; // 4. อัปเดตคิว (เอาเฉพาะตัวที่ถูกสุ่มตัวที่ 3 ออกจากคิว) let newQueue = currentQueue.filter(item => item !== thirdAsset); await db.collection("settings").doc("queue").set({ list: newQueue }); // 5. บันทึกข้อมูลลง Firebase await db.collection("posts").add({ date: dateVal, kw: { m: {name: chosen[0], num: 0}, n: {name: chosen[1], num: 0}, e: {name: chosen[2], num: 0} }, su: { m: {name: chosen[1], num: 0}, n: {name: chosen[2], num: 0}, e: {name: chosen[0], num: 0} }, timestamp: firebase.firestore.FieldValue.serverTimestamp() }); alert("🎲 สุ่มทรัพย์ตัวที่ 3 สำเร็จ! (ล็อคบ้านเอื้อ+Happy เรียบร้อย)"); checkTodayStatus(); loadHistory(); } catch (error) { alert("เกิดข้อผิดพลาด: " + error.message); } finally { btn.disabled = false; btn.innerText = "🎲 กดสุ่มทรัพย์ (ตัวที่ 3)"; } }
