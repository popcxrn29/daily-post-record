async function startRandom() { const btn = document.querySelector('.btn-random'); const dateVal = document.getElementById("date").value; if (!dateVal) { alert("กรุณาเลือกวันที่ก่อนครับ"); return; } try { btn.disabled = true; btn.innerText = "กำลังสุ่ม..."; // 1. ดึงคิวทรัพย์ที่เหลือ let qSnap = await db.collection("settings").doc("queue").get(); let currentQueue = (qSnap.exists && qSnap.data().list && qSnap.data().list.length >= 3) ? qSnap.data().list : [...allAssets]; // 2. สุ่มเลือก 3 ทรัพย์ (แบบ Simple Random เพื่อลดการค้าง) let chosen = []; for (let i = 0; i < 3; i++) { let idx = Math.floor(Math.random() * currentQueue.length); chosen.push(currentQueue.splice(idx, 1)[0]); } // 3. บันทึกคิวที่เหลือกลับไป await db.collection("settings").doc("queue").set({ list: currentQueue }); // 4. บันทึกข้อมูลลงในประวัติ await db.collection("posts").add({ date: dateVal, kw: { m: {name: chosen[0], num: ""}, n: {name: chosen[1], num: ""}, e: {name: chosen[2], num: ""} }, su: { m: {name: chosen[1], num: ""}, n: {name: chosen[2], num: ""}, e: {name: chosen[0], num: ""} }, timestamp: firebase.firestore.FieldValue.serverTimestamp() }); alert("สุ่มและบันทึกข้อมูลสำเร็จ!"); // รีโหลดหน้าจอเพื่อแสดงผล location.reload(); } catch (error) { alert("เกิดข้อผิดพลาด: " + error.message); console.error(error); } finally { btn.disabled = false; btn.innerText = "🎲 กดสุ่มทรัพย์"; } }
